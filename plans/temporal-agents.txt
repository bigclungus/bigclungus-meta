4. Workflow (workflows/agent_task_workflow.py)
python


from datetime import timedelta
from temporalio import workflow
from temporalio.exceptions import ActivityError, CancelledError
from temporalio.common import RetryPolicy
from ..types import AgentTaskInput
from ..activities.task_db import create_task_record, finalize_task, record_error
from ..activities.agent_executor import execute_agent
@workflow.defn(name="AgentTaskWorkflow")
class AgentTaskWorkflow:
    def __init__(self) -> None:
        self._status = "running"
        self._metadata: dict = {}
        self._result: dict | None = None
    @workflow.run
    async def run(self, input: AgentTaskInput) -> dict:
        await workflow.execute_local_activity(
            create_task_record,
            input,
            start_to_close_timeout=timedelta(seconds=10),
        )
        try:
            result = await workflow.execute_activity(
                execute_agent,
                input,
                start_to_close_timeout=timedelta(minutes=90),
                heartbeat_timeout=timedelta(seconds=90),
                retry_policy=RetryPolicy(maximum_attempts=1),
            )
            self._result = result
            self._status = "completed"
        except ActivityError as e:
            if isinstance(e.cause, CancelledError) or "cancelled" in str(e).lower():
                self._status = "cancelled"
            else:
                self._status = "failed"
                await workflow.execute_local_activity(
                    record_error, [input.task_id, str(e)]
                )
            raise
        finally:
            await workflow.execute_local_activity(
                finalize_task,
                [input, self._result or {"status": self._status}],
                start_to_close_timeout=timedelta(seconds=30),
            )
        return self._result or {}
    @workflow.signal
    async def add_metadata(self, data: dict) -> None:
        self._metadata.update(data)
    @workflow.signal
    async def cancel(self) -> None:
        self._status = "cancelled"
    @workflow.query
    def get_status(self) -> dict:
        return {
            "task_id": workflow.info().workflow_id,
            "status": self._status,
            "metadata": self._metadata,
            "result": self._result,
        }

import asyncio
import os
from temporalio import activity
from temporalio.exceptions import CancelledError
from ..types import AgentTaskInput
@activity.defn
async def execute_agent(input: AgentTaskInput) -> dict:
    if activity.is_cancelled():
        raise CancelledError()
    cmd = ["claude", "-p", input.prompt, "--model", input.model]
    if input.extra_cli_args:
        cmd.extend(input.extra_cli_args)
    env = os.environ.copy()
    if input.api_key:
        env["ANTHROPIC_API_KEY"] = input.api_key
        cmd.extend(["--api-key", input.api_key])
    proc = await asyncio.create_subprocess_exec(
        *cmd, env=env, stdout=asyncio.subprocess.PIPE, stderr=asyncio.subprocess.PIPE
    )
    heartbeat_task = asyncio.create_task(_heartbeat_loop(proc, input.task_id))
    try:
        stdout, stderr = await proc.communicate()
        if proc.returncode != 0:
            raise RuntimeError(f"claude exited {proc.returncode}: {stderr.decode()}")
        return _parse_output(stdout.decode(), input.model)
    finally:
        if not heartbeat_task.done():
            heartbeat_task.cancel()
async def _heartbeat_loop(proc, task_id: str):
    while True:
        await asyncio.sleep(25)
        activity.heartbeat({"task_id": task_id, "status": "running", "pid": proc.pid})
def _parse_output(output: str, model: str) -> dict:
    return {
        "status": "success",
        "model": model,
        "output": output.strip(),
        "input_tokens": 0,   # replace with real parsing
        "output_tokens": 0,
        "cost_usd": 0.0,
        "raw": output,
    }

6. Control Flow Examples
Example 1: Starting a Foreground Subagent (Happy Path)

subagent-start.ts receives prompt + context (model, api_key, discord_user, etc.).
Generates task_id and workflowId = "agent-task-{task_id}".
Calls Temporal HTTP API to start AgentTaskWorkflow with AgentTaskInput (includes is_foreground: true, model, api_key, metadata).
Workflow executes:
create_task_record (Local Activity) → writes initial row to tasks.db and agents.db.
execute_agent (Activity with heartbeat) → runs claude --model {model} -p "{prompt}" --api-key {key}.
subagent-start.ts also performs legacy DB write + clunger POST (shadow mode).
Hook writes workflow reference to /tmp/subagent-{task_id}.json.
Example 2: During Execution (Metadata Update + Cancellation)

External code (or subagent-stop.ts) calls:
bash


POST /api/v1/namespaces/tasks/workflows/agent-task-xxx/signal/add_metadata
{ "input": [{"progress": "thinking", "partial_tokens": 1234}] }
Workflow receives add_metadata signal → updates self._metadata.
To cancel: send cancel signal → workflow sets _status = "cancelled".
execute_agent Activity checks activity.is_cancelled() on each heartbeat and terminates the claude subprocess cleanly.
Example 3: Completion & Finalization

execute_agent finishes → returns usage dict.
Workflow runs finalize_task (Local Activity) → updates both DBs with final status, tokens, cost, finished_at, and appends to history.
subagent-stop.ts loads workflow reference from /tmp and sends one final add_metadata signal with last message.
Workflow completes. Result is permanently stored in Temporal history (infinite retention).
Example 4: Querying / Getting Back Data

UI calls GET /api/tasks/temporal?model=claude-3-5-sonnet...&is_foreground=true → Temporal Visibility API using Search Attributes.
Or direct query on a running workflow: client.get_workflow_handle(workflow_id).query(AgentTaskWorkflow.get_status).

async function startForegroundSubagent(context: any): Promise<string> {
  const taskId = `fg-${Date.now()}-${Math.random().toString(36).slice(2,8)}`;
  const workflowId = `agent-task-${taskId}`;
  const input = {
    task_id: taskId,
    model: context.model || "claude-3-5-sonnet-20240620",
    prompt: context.prompt,
    api_key: context.apiKey,
    is_foreground: true,
    metadata: { discord_user: context.discordUser, ...context.extra }
  };
  // Temporal start (shadow)
  await fetch(`http://127.0.0.1:8233/api/v1/namespaces/tasks/workflows`, {
    method: "POST",
    body: JSON.stringify({ workflow_id: workflowId, workflow_type: "AgentTaskWorkflow", task_queue: "agent-tasks-queue", input: [input] })
  }).catch(() => {});   // never break hook
  // save reference + legacy path
  writeFileSync(`/tmp/subagent-${taskId}.json`, JSON.stringify({workflow_id: workflowId}));
  await legacyWrite(taskId, input);
  return taskId;
}
